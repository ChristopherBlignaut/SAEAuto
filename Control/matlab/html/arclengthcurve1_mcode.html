1function [scoefx, scoefy, si] = arclengthcurve(points, gran, epsilon)
 2%#codegen
 3%UNTITLED13 Summary of this function goes here
 4%   Detailed explanation goes here
 5
 6%epsilon = 1e-4; %Acceptable error in algorithms.
 7
 8%build spline from points using minimal tangent acceleration
 9ti = cumsum([0;((diff(points.').^2)*ones(2,1)).^(1/4)]);
10[coefx, coefy] = cubic_spline3(points(1,:)',points(2,:)',ti);
11
12nsegments = length(ti)-1;
13lengths = zeros(nsegments,1);
14%use adaptive gaussian to calculate length of each segment
15for n=1:nsegments
16    lengths(n) = adaptivegaus(coefx(n,:),coefy(n,:),ti(n),ti(n+1),ti(n),epsilon);
17end
18%find the total length of the original curve.
19totallength = sum(lengths);
20
21%now build new curve
22
23%now compute m+1 equally spaced points on original curvne
24%so the length of each segment should be
25nsegmentsnew = ceil(totallength/gran);
26l = totallength/nsegmentsnew;
27
28index = 1; %index of original curv segments
29tihat = zeros(nsegmentsnew+1,1); %initialize tihat vector
30newpoints = zeros(2,nsegmentsnew + 1); %initialize new points matrix
31newpoints(1,1) = evalspline(coefx(1,:),0,0,0); %evaluate x at 0
32newpoints(2,1) = evalspline(coefy(1,:),0,0,0); %evaluate y at 0
33si = zeros(nsegmentsnew+1,1); %initialize si vector
34%start at j=2 because si,tihat at j=1 =0.
35j=2;
36cumulength = lengths(1); %cumulative length at first iteration
37searchlength = ((j-1)*l); %desired length at first iteration
38
39while j <= nsegmentsnew + 1
40    if cumulength >= searchlength - epsilon
41        %found our segment to look for tihat, now use bisection
42        si(j) = searchlength;
43        len = si(j) - (cumulength - lengths(index));
44        if len < epsilon
45            tihat(j) = cumulength;
46        else
47            tihat(j) =  bisect(ti(index),ti(index+1),...
48                len,coefx(index,:),coefy(index,:), epsilon);
49        end
50        
51        newpoints(1,j) = evalspline(coefx(index,:),...
52            tihat(j),ti(index),0);
53        newpoints(2,j) = evalspline(coefy(index,:),...
54            tihat(j),ti(index),0);
55        
56        %set up next search
57        j = j+1;
58        searchlength = ((j-1)*l);
59    else
60        %tihat not in this curve index so increment index
61        index = index + 1;
62        cumulength = cumulength + lengths(index);
63    end
64end
65[scoefx, scoefy] = cubic_spline3(newpoints(1,:)',newpoints(2,:)',si);
66
67end
