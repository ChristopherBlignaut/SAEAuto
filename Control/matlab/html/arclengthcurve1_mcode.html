1function [scoefx, scoefy, si] = arclengthcurve(points, nsegmentsnew)
 2%#codegen
 3%UNTITLED13 Summary of this function goes here
 4%   Detailed explanation goes here
 5
 6epsilon = 1e-4; %Acceptable error in algorithms.
 7%neqpoints = c; %number of points to split the arc length curve into.
 8
 9%build spline from points using minimal tangent acceleration
10ti = cumsum([0;((diff(points.').^2)*ones(2,1)).^(1/4)]);
11[coefx, coefy] = cubic_spline3(points(1,:)',points(2,:)',ti);
12
13nsegments = length(ti)-1;
14lengths = zeros(nsegments,1);
15%use adaptive gaussian to calculate length of each segment
16for n=1:nsegments
17    lengths(n) = adaptivegaus(coefx(n,:),coefy(n,:),ti(n),ti(n+1),ti(n),epsilon);
18end
19%find the total length of the original curve.
20totallength = sum(lengths);
21
22%now build new curve
23
24%nsegmentsnew = 30; %number of new segments
25
26%now compute m+1 equally spaced points on original curvne
27%so the length of each segment should be
28l = totallength/nsegmentsnew;
29
30index = 1; %index of original curv segments
31tihat = zeros(nsegmentsnew+1,1); %initialize tihat vector
32newpoints = zeros(2,nsegmentsnew + 1); %initialize new points matrix
33newpoints(1,1) = evalspline(coefx(1,:),0,0,0); %evaluate x at 0
34newpoints(2,1) = evalspline(coefy(1,:),0,0,0); %evaluate y at 0
35si = zeros(nsegmentsnew+1,1); %initialize si vector
36%start at j=2 because si,tihat at j=1 =0.
37j=2;
38cumulength = lengths(1); %cumulative length at first iteration
39searchlength = ((j-1)*l); %desired length at first iteration
40
41while j <= nsegmentsnew + 1
42    if cumulength >= searchlength - epsilon
43        %found our segment to look for tihat, now use bisection
44        si(j) = searchlength;
45        len = si(j) - (cumulength - lengths(index));
46        if len < epsilon
47            tihat(j) = cumulength;
48        else
49            tihat(j) =  bisect(ti(index),ti(index+1),...
50                len,coefx(index,:),coefy(index,:), epsilon);
51        end
52        
53        newpoints(1,j) = evalspline(coefx(index,:),...
54            tihat(j),ti(index),0);
55        newpoints(2,j) = evalspline(coefy(index,:),...
56            tihat(j),ti(index),0);
57        
58        %set up next search
59        j = j+1;
60        searchlength = ((j-1)*l);
61    else
62        %tihat not in this curve index so increment index
63        index = index + 1;
64        cumulength = cumulength + lengths(index);
65    end
66end
67[scoefx, scoefy] = cubic_spline3(newpoints(1,:)',newpoints(2,:)',si);
68
69end
